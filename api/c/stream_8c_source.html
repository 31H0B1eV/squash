153 static int
154 squash_stream_thread_func (SquashStream* stream) {
155  SquashStreamPrivate* priv = stream->priv;
156  SquashOperation operation;
157 
158  mtx_lock (&(priv->io_mtx));
159  priv->result = SQUASH_OK;
160  cnd_signal (&(priv->result_cnd));
161 
162  while ((operation = priv->request) == SQUASH_OPERATION_INVALID) {
163  cnd_wait (&(priv->request_cnd), &(priv->io_mtx));
164  }
165  priv->request = SQUASH_OPERATION_INVALID;
166 
167  priv->result = stream->codec->funcs.process_stream (stream, operation);
168  priv->finished = true;
169  cnd_signal (&(priv->result_cnd));
170  mtx_unlock (&(priv->io_mtx));
171 
172  return 0;
173 }

205 static SquashStatus
206 squash_stream_send_to_thread (SquashStream* stream, SquashOperation operation) {
207  SquashStreamPrivate* priv = stream->priv;
208  SquashStatus result;
209 
210  priv->request = operation;
211  cnd_signal (&(priv->request_cnd));
212  mtx_unlock (&(priv->io_mtx));
213 
214  mtx_lock (&(priv->io_mtx));
215  while ((result = priv->result) == SQUASH_STATUS_INVALID) {
216  cnd_wait (&(priv->result_cnd), &(priv->io_mtx));
217  }
218  priv->result = SQUASH_STATUS_INVALID;
219 
220  if (priv->finished == true) {
221  mtx_unlock (&(priv->io_mtx));
222  thrd_join (priv->thread, NULL);
223  }
224 
225  return result;
226 }

241 squash_stream_init (void* stream,
242  SquashCodec* codec,
243  SquashStreamType stream_type,
244  SquashOptions* options,
245  SquashDestroyNotify destroy_notify) {
246  SquashStream* s;
247 
248  assert (stream != NULL);
249 
250  s = (SquashStream*) stream;
251 
252  squash_object_init (stream, false, destroy_notify);
253 
254  s->next_in = NULL;
255  s->avail_in = 0;
256  s->total_in = 0;
257 
258  s->next_out = NULL;
259  s->avail_out = 0;
260  s->total_out = 0;
261 
262  s->codec = codec;
263  s->options = (options != NULL) ? squash_object_ref (options) : NULL;
264  s->stream_type = stream_type;
265  s->state = SQUASH_STREAM_STATE_IDLE;
266 
267  s->user_data = NULL;
268  s->destroy_user_data = NULL;
269 
270  if (SQUASH_UNLIKELY((codec->funcs.info & SQUASH_CODEC_INFO_RUN_IN_THREAD) == SQUASH_CODEC_INFO_RUN_IN_THREAD)) {
271  s->priv = malloc (sizeof (SquashStreamPrivate));
272 
273  mtx_init (&(s->priv->io_mtx), mtx_plain);
274  mtx_lock (&(s->priv->io_mtx));
275 
276  s->priv->request = SQUASH_OPERATION_INVALID;
277  cnd_init (&(s->priv->request_cnd));
278 
279  s->priv->result = SQUASH_STATUS_INVALID;
280  cnd_init (&(s->priv->result_cnd));
281 
282  s->priv->finished = false;
283  int res = thrd_create (&(s->priv->thread), (thrd_start_t) squash_stream_thread_func, s);
284  assert (res == thrd_success);
285 
286  while (s->priv->result == SQUASH_STATUS_INVALID)
287  cnd_wait (&(s->priv->result_cnd), &(s->priv->io_mtx));
288  s->priv->result = SQUASH_STATUS_INVALID;
289  } else {
290  s->priv = NULL;
291  }
292 }

305 
306  assert (stream != NULL);
307 
308  s = (SquashStream*) stream;
309 
310  if (SQUASH_UNLIKELY(s->priv != NULL)) {

312 
313  if (!priv->finished) {
314  thrd_exit (-1);
315  }
316  cnd_destroy (&(priv->request_cnd));
317  cnd_destroy (&(priv->result_cnd));
318  mtx_destroy (&(priv->io_mtx));
319 
320  free (s->priv);
321  }
322 
323  if (s->destroy_user_data != NULL && s->user_data != NULL) {
324  s->destroy_user_data (s->user_data);
325  }
326 
327  if (s->options != NULL) {
328  s->options = squash_object_unref (s->options);
329  }
330 
331  squash_object_destroy (stream);
332 }

423 static SquashStatus
424 squash_stream_process_internal (SquashStream* stream, SquashOperation operation) {
425  SquashCodec* codec;
426  SquashCodecFuncs* funcs = NULL;
427  SquashStatus res;
428  SquashOperation current_operation;
429 
430  assert (stream != NULL);
431  codec = stream->codec;
432  assert (codec != NULL);
433  funcs = squash_codec_get_funcs (codec);
434  assert (funcs != NULL);
435 
436  /* Flush is optional, so return an error if it doesn't exist but
437  flushing was requested. */
438  if (operation == SQUASH_OPERATION_FLUSH && ((funcs->info & SQUASH_CODEC_INFO_CAN_FLUSH) == 0)) {
439  return SQUASH_INVALID_OPERATION;
440  }
441 
442  /* In order to take some of the load off of the plugins, there is
443  some extra logic here which may seem a bit disorienting at first
444  glance. Basically, instead of requiring that plugins handle
445  flushing or finishing with arbitrarily large inputs, we first try
446  to process as much input as we can. So, when someone calls
447  squash_stream_flush or squash_stream finish Squash may, depending
448  on the stream state, first call the process function. Note that
449  Squash will not flush a stream before finishing it (unless there
450  is logic to do so in the plugin) as it could cause an increase in
451  the output size (it does with zlib).
452 
453  One interesting consequence of this is that the stream_state
454  field may not be what you're expecting. If an earlier operation
455  returned SQUASH_PROCESSING, stream_type may never transition to
456  the new value. In this case, the stream_type does accurately
457  represent the state of the stream, though it probably isn't wise
458  to depend on that behavior. */
459 
460  if ((operation == SQUASH_OPERATION_PROCESS && stream->state > SQUASH_STREAM_STATE_RUNNING) ||
461  (operation == SQUASH_OPERATION_FLUSH && stream->state > SQUASH_STREAM_STATE_FLUSHING) ||
462  (operation == SQUASH_OPERATION_FINISH && stream->state > SQUASH_STREAM_STATE_FINISHING)) {
463  return SQUASH_STATE;
464  }
465 
466  switch (stream->state) {
467  case SQUASH_STREAM_STATE_IDLE:
468  case SQUASH_STREAM_STATE_RUNNING:
469  current_operation = SQUASH_OPERATION_PROCESS;
470  break;
471  case SQUASH_STREAM_STATE_FLUSHING:
472  current_operation = SQUASH_OPERATION_FLUSH;
473  break;
474  case SQUASH_STREAM_STATE_FINISHING:
475  current_operation = SQUASH_OPERATION_FINISH;
476  break;
477  case SQUASH_STREAM_STATE_FINISHED:
478  current_operation = (SQUASH_OPERATION_FINISH + 1);
479  break;
480  }
481 
482  if (current_operation > operation) {
483  return SQUASH_STATE;
484  }
485 
486  const size_t avail_in = stream->avail_in;
487  const size_t avail_out = stream->avail_out;
488 
489  /* Some libraries (like zlib) will realize that we're not providing
490  it any room for output and are eager to tell us that we don't
491  have any space instead of decoding the stream enough to know if we
492  actually need that space.
493 
494  In cases where this might be problematic, we provide a
495  single-byte buffer to the plugin instead. If anything actually
496  gets written to it then we'll return an error
497  (SQUASH_BUFFER_FULL), which is non-recoverable.
498 
499  There are a few cases where this might reasonably be a problem:
500 
501  * Decompression streams which know the exact size of the
502  decompressed output, when using codecs which contain extra
503  data at the end, such as a footer or EOS marker.
504 
505  * Compression streams writing to a fixed buffer with a length of
506  less than or equal to max_compressed_size bytes. This is a
507  pretty reasonable thing to do, since you might want to only
508  bother using compression if you can achieve a certain ratio.
509 
510  For consumers which don't satisfy either of these conditions,
511  this code should never be reached. */
512 
513  uint8_t* next_out = NULL;
514  uint8_t output_sbb = 0;
515  if (stream->avail_out == 0) {
516  next_out = stream->next_out;
517  stream->avail_out = 1;
518  stream->next_out = &output_sbb;
519  }
520 
521  while (current_operation <= operation) {
522  if (current_operation == SQUASH_OPERATION_PROCESS) {
523  /* Process */
524  if (stream->avail_in == 0 && stream->state == SQUASH_STREAM_STATE_IDLE) {
525  res = SQUASH_OK;
526  } else if ((funcs->info & SQUASH_CODEC_INFO_RUN_IN_THREAD) == SQUASH_CODEC_INFO_RUN_IN_THREAD) {
527  res = squash_stream_send_to_thread (stream, current_operation);
528  } else if (funcs->process_stream != NULL) {
529  res = funcs->process_stream (stream, current_operation);
530  } else {
531  res = squash_buffer_stream_process ((SquashBufferStream*) stream);
532  }
533  } else if (current_operation == SQUASH_OPERATION_FLUSH) {
534  /* Flush */
535  if (current_operation == operation) {
536  if ((funcs->info & SQUASH_CODEC_INFO_CAN_FLUSH) == SQUASH_CODEC_INFO_CAN_FLUSH) {
537  if ((funcs->info & SQUASH_CODEC_INFO_RUN_IN_THREAD) == SQUASH_CODEC_INFO_RUN_IN_THREAD) {
538  res = squash_stream_send_to_thread (stream, current_operation);
539  } else if (funcs->process_stream == NULL) {
540  return SQUASH_INVALID_OPERATION;
541  } else {
542  res = funcs->process_stream (stream, current_operation);
543  }
544  } else {
545  /* We aready checked to make sure the stream is flushable if
546  the user called flush directly, so if this code is
547  reached the user didn't call flush, they called finish
548  which attempts to flush internally. Just pretend it
549  worked so we can proceed to finishing. */
550  res = SQUASH_OK;
551  }
552  }
553  } else if (current_operation == SQUASH_OPERATION_FINISH) {
554  /* Finish */
555  if (funcs->process_stream != NULL) {
556  if ((funcs->info & SQUASH_CODEC_INFO_RUN_IN_THREAD) == SQUASH_CODEC_INFO_RUN_IN_THREAD) {
557  res = squash_stream_send_to_thread (stream, current_operation);
558  } else {
559  res = funcs->process_stream (stream, current_operation);
560  }
561  } else {
562  res = squash_buffer_stream_finish ((SquashBufferStream*) stream);
563  }
564 
565  /* Plugins *should* return SQUASH_OK, not SQUASH_END_OF_STREAM,
566  from the finish function, but it's an easy mistake to make
567  (and correct), so... */
568  if (SQUASH_UNLIKELY(res == SQUASH_END_OF_STREAM)) {
569  res = SQUASH_OK;
570  }
571  }
572 
573  /* Check our internal single byte buffer */
574  if (next_out != 0) {
575  if (stream->avail_out == 0) {
576  res = SQUASH_BUFFER_FULL;
577  }
578  }
579 
580  if (res == SQUASH_PROCESSING) {
581  switch (current_operation) {
582  case SQUASH_OPERATION_PROCESS:
583  stream->state = SQUASH_STREAM_STATE_RUNNING;
584  break;
585  case SQUASH_OPERATION_FLUSH:
586  stream->state = SQUASH_STREAM_STATE_FLUSHING;
587  break;
588  case SQUASH_OPERATION_FINISH:
589  stream->state = SQUASH_STREAM_STATE_FINISHING;
590  break;
591  }
592  break;
593  } else if (res == SQUASH_END_OF_STREAM || (current_operation == SQUASH_OPERATION_FINISH && res == SQUASH_OK)) {
594  stream->state = SQUASH_STREAM_STATE_FINISHED;
595  current_operation++;
596  break;
597  } else if (res == SQUASH_OK) {
598  stream->state = SQUASH_STREAM_STATE_IDLE;
599  current_operation++;
600  } else {
601  break;
602  }
603  }
604 
605  if (next_out != 0) {
606  stream->avail_out = 0;
607  stream->next_out = next_out;
608  }
609 
610  stream->total_in += (avail_in - stream->avail_in);
611  stream->total_out += (avail_out - stream->avail_out);
612 
613  return res;
614 }
