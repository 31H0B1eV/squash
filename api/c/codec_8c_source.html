486 SquashStatus
487 squash_codec_compress_with_options (SquashCodec* codec,
488  uint8_t* compressed, size_t* compressed_length,
489  const uint8_t* uncompressed, size_t uncompressed_length,
490  SquashOptions* options) {
491  SquashPlugin* plugin = NULL;
492  SquashCodecFuncs* funcs = NULL;
493 
494  assert (codec != NULL);
495 
496  plugin = codec->plugin;
497  assert (plugin != NULL);
498 
499  funcs = squash_codec_get_funcs (codec);
500  if (funcs == NULL) {
501  return SQUASH_UNABLE_TO_LOAD;
502  } else if (funcs->compress_buffer != NULL) {
503  return funcs->compress_buffer (codec,
504  compressed, compressed_length,
505  uncompressed, uncompressed_length,
506  options);
507  } else {
508  SquashStatus status;
509  SquashStream* stream;
510 
511  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_COMPRESS, options);
512  stream->next_in = uncompressed;
513  stream->avail_in = uncompressed_length;
514  stream->next_out = compressed;
515  stream->avail_out = *compressed_length;
516 
517  do {
518  status = squash_stream_process (stream);
519  } while (status == SQUASH_PROCESSING);
520 
521  if (status != SQUASH_OK) {
522  squash_object_unref (stream);
523  return status;
524  }
525 
526  do {
527  status = squash_stream_finish (stream);
528  } while (status == SQUASH_PROCESSING);
529 
530  if (status != SQUASH_OK) {
531  squash_object_unref (stream);
532  return status;
533  }
534 
535  *compressed_length = stream->total_out;
536  squash_object_unref (stream);
537  return status;
538  }
539 }

600  if (funcs == NULL) {
601  return SQUASH_UNABLE_TO_LOAD;
602  } else if (funcs->decompress_buffer != NULL) {
603  return funcs->decompress_buffer (codec,
604  decompressed, decompressed_length,
605  compressed, compressed_length,
606  options);
607  } else {
608  SquashStatus status;
609  SquashStream* stream;
610 
611  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_DECOMPRESS, options);
612  stream->next_in = compressed;

614  stream->next_out = decompressed;
615  stream->avail_out = *decompressed_length;
616 
617  do {
618  status = squash_stream_finish (stream);
619  } while (status == SQUASH_PROCESSING);
620 
621  switch (status) {
622  case SQUASH_END_OF_STREAM:
623  status = SQUASH_OK;
624  case SQUASH_OK:
625  *decompressed_length = stream->total_out;
626  default:
627  squash_object_unref (stream);
628  }
629 
630  return status;
631  }

859 static SquashStatus
860 squash_codec_process_file_with_options (SquashCodec* codec,
861  SquashStreamType stream_type,
862  FILE* output, FILE* input,
863  SquashOptions* options) {
864  SquashStatus res = SQUASH_FAILED;
865  SquashStream* stream;
866  uint8_t* inbuf = NULL;
867  uint8_t* outbuf = NULL;
868 
869 #ifdef _WIN32
870  /* TODO: MapViewOfFile (Windows) version. */
871 #else
872  if (((stream_type == SQUASH_STREAM_COMPRESS) && codec->funcs.compress_buffer != NULL) ||
873  ((stream_type == SQUASH_STREAM_DECOMPRESS) && codec->funcs.decompress_buffer != NULL)) {
874  /* Attempt to mmap the input and output. This short circuits the
875  whole SquashBufferStream hack when possible, which can be a
876  huge performance boost (50-100% for several codecs). */
877  struct stat instat = { 0, };
878  size_t outsize_max = 0;
879  int infd = fileno (input);
880  int outfd = fileno (output);
881 
882  if (infd > 0) {
883  if (fstat (infd, &instat) != -1) {
884  if ((inbuf = mmap (NULL, instat.st_size, PROT_READ, MAP_SHARED, infd, 0)) != MAP_FAILED) {
885  if (stream_type == SQUASH_STREAM_COMPRESS) {
886  outsize_max = squash_codec_get_max_compressed_size (codec, instat.st_size);
887  } else if (codec->funcs.get_uncompressed_size != NULL) {
888  outsize_max = squash_codec_get_uncompressed_size (codec, inbuf, instat.st_size);
889  } else {
890  outsize_max = instat.st_size - 1;
891  outsize_max |= outsize_max >> 1;
892  outsize_max |= outsize_max >> 2;
893  outsize_max |= outsize_max >> 4;
894  outsize_max |= outsize_max >> 8;
895  outsize_max |= outsize_max >> 16;
896  outsize_max++;
897 
898  outsize_max <<= 1;
899  outsize_max *= 2;
900  }
901 
902  if (ftruncate(outfd, (off_t) outsize_max) != -1) {
903  if ((outbuf = mmap (NULL, outsize_max, PROT_WRITE | PROT_READ, MAP_SHARED, outfd, 0)) != MAP_FAILED) {
904  size_t outsize = outsize_max;
905 
906  if (stream_type == SQUASH_STREAM_COMPRESS) {
907  res = squash_codec_compress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
908 
909  if (res == SQUASH_OK) {
910  ftruncate (outfd, (off_t) outsize);
911  fseek (output, outsize, SEEK_SET);
912  }
913  } else {
914  res = squash_codec_decompress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
915  while (res == SQUASH_BUFFER_FULL) {
916  munmap (outbuf, outsize_max);
917  outsize_max <<= 1;
918  if ((ftruncate (outfd, (off_t) outsize_max) == -1) ||
919  (outbuf = mmap (NULL, outsize_max, PROT_WRITE | PROT_READ, MAP_SHARED, outfd, 0)) == MAP_FAILED) {
920  break;
921  } else {
922  outsize = outsize_max;
923  res = squash_codec_decompress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
924  }
925  }
926 
927  if (res == SQUASH_OK) {
928  ftruncate (outfd, (off_t) outsize);
929  }
930  }
931 
932  if (outbuf != MAP_FAILED) {
933  munmap (outbuf, outsize_max);
934  }
935  }
936  }
937 
938  munmap (inbuf, instat.st_size);
939  }
940  }
941  }
942  }
943 #endif
944 
945  if (res == SQUASH_OK) {
946  return res;
947  }
948 
949  stream = squash_codec_create_stream_with_options (codec, stream_type, options);
950  if ( stream == NULL ) {
951  return SQUASH_FAILED;
952  }
953 
954  inbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
955  outbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
956 
957  if (inbuf == NULL || outbuf == NULL) {
958  squash_object_unref (stream);
959  free (inbuf);
960  free (outbuf);
961  return SQUASH_MEMORY;
962  }
963 
964  while ( !feof (input) ) {
965  stream->next_in = inbuf;
966  stream->avail_in = fread (inbuf, 1, SQUASH_CODEC_FILE_BUF_SIZE, input);
967 
968  do {
969  stream->next_out = outbuf;
970  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
971 
972  res = squash_stream_process (stream);
973 
974  fwrite (outbuf, 1, stream->next_out - outbuf, output);
975  } while ( res == SQUASH_PROCESSING );
976 
977  if ( res != SQUASH_OK ) {
978  if ( res == SQUASH_END_OF_STREAM && stream_type == SQUASH_STREAM_DECOMPRESS ) {
979  res = SQUASH_OK;
980  }
981  squash_object_unref (stream);
982  free (inbuf);
983  free (outbuf);
984  return res;
985  }
986  }
987 
988  do {
989  stream->next_out = outbuf;
990  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
991 
992  res = squash_stream_finish (stream);
993 
994  fwrite (outbuf, 1, stream->next_out - outbuf, output);
995  } while ( res == SQUASH_PROCESSING );
996 
997  squash_object_unref (stream);
998  free (inbuf);
999  free (outbuf);
1000 
1001  return res;
1002 }
