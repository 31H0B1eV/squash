485 SquashStatus
486 squash_codec_compress_with_options (SquashCodec* codec,
487  uint8_t* compressed, size_t* compressed_length,
488  const uint8_t* uncompressed, size_t uncompressed_length,
489  SquashOptions* options) {
490  SquashPlugin* plugin = NULL;
491  SquashCodecFuncs* funcs = NULL;
492 
493  assert (codec != NULL);
494 
495  plugin = codec->plugin;
496  assert (plugin != NULL);
497 
498  assert (compressed != NULL);
499  assert (uncompressed != NULL);
500 
501  funcs = squash_codec_get_funcs (codec);
502  if (funcs == NULL)
503  return SQUASH_UNABLE_TO_LOAD;
504 
505  if (compressed == uncompressed)
506  return SQUASH_INVALID_BUFFER;
507 
508  if (funcs->compress_buffer != NULL) {
509  return funcs->compress_buffer (codec,
510  compressed, compressed_length,
511  uncompressed, uncompressed_length,
512  options);
513  } else {
514  SquashStatus status;
515  SquashStream* stream;
516 
517  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_COMPRESS, options);
518  stream->next_in = uncompressed;
519  stream->avail_in = uncompressed_length;
520  stream->next_out = compressed;
521  stream->avail_out = *compressed_length;
522 
523  do {
524  status = squash_stream_process (stream);
525  } while (status == SQUASH_PROCESSING);
526 
527  if (status != SQUASH_OK) {
528  squash_object_unref (stream);
529  return status;
530  }
531 
532  do {
533  status = squash_stream_finish (stream);
534  } while (status == SQUASH_PROCESSING);
535 
536  if (status != SQUASH_OK) {
537  squash_object_unref (stream);
538  return status;
539  }
540 
541  *compressed_length = stream->total_out;
542  squash_object_unref (stream);
543  return status;
544  }
545 }

593 squash_codec_decompress_with_options (SquashCodec* codec,
594  uint8_t* decompressed, size_t* decompressed_length,
595  const uint8_t* compressed, size_t compressed_length,
596  SquashOptions* options) {
597  SquashPlugin* plugin = NULL;

599 
600  assert (codec != NULL);
601 
602  plugin = codec->plugin;
603  assert (plugin != NULL);
604 
605  funcs = squash_codec_get_funcs (codec);
606  if (funcs == NULL)
607  return SQUASH_UNABLE_TO_LOAD;
608 
609  if (decompressed == compressed)
610  return SQUASH_INVALID_BUFFER;
611 
612  if (funcs->decompress_buffer != NULL) {
613  return funcs->decompress_buffer (codec,
614  decompressed, decompressed_length,
615  compressed, compressed_length,
616  options);
617  } else {
618  SquashStatus status;
619  SquashStream* stream;
620 
621  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_DECOMPRESS, options);
622  stream->next_in = compressed;
623  stream->avail_in = compressed_length;
624  stream->next_out = decompressed;
625  stream->avail_out = *decompressed_length;
626 
627  do {
628  status = squash_stream_process (stream);
629  } while (status == SQUASH_PROCESSING);
630 
631  if (status == SQUASH_END_OF_STREAM) {
632  status = SQUASH_OK;
633  *decompressed_length = stream->total_out;
634  } else if (status == SQUASH_OK) {
635  do {
636  status = squash_stream_finish (stream);
637  } while (status == SQUASH_PROCESSING);
638 
639  if (status == SQUASH_OK) {
640  *decompressed_length = stream->total_out;
641  }
642  }
643 
644  assert (stream->stream_type == SQUASH_STREAM_DECOMPRESS);
645  squash_object_unref (stream);
646 
647  return status;
648  }
649 }

876 static SquashStatus
877 squash_codec_process_file_with_options (SquashCodec* codec,
878  SquashStreamType stream_type,
879  FILE* output, FILE* input,
880  SquashOptions* options) {
881  SquashStatus res = SQUASH_FAILED;
882  SquashStream* stream;
883  uint8_t* inbuf = NULL;
884  uint8_t* outbuf = NULL;
885 
886 #if !defined(_WIN32)
887  if (((stream_type == SQUASH_STREAM_COMPRESS) && codec->funcs.compress_buffer != NULL) ||
888  ((stream_type == SQUASH_STREAM_DECOMPRESS) && codec->funcs.decompress_buffer != NULL)) {
889  /* Attempt to mmap the input and output. This short circuits the
890  whole SquashBufferStream hack when possible, which can be a
891  huge performance boost (50-100% for several codecs). */
892  struct stat instat = { 0, };
893  size_t outsize_max = 0;
894  int infd = fileno (input);
895  int outfd = fileno (output);
896 
897  if (infd > 0) {
898  if (fstat (infd, &instat) != -1) {
899  if ((inbuf = mmap (NULL, instat.st_size, PROT_READ, MAP_SHARED, infd, 0)) != MAP_FAILED) {
900  if (stream_type == SQUASH_STREAM_COMPRESS) {
901  outsize_max = squash_codec_get_max_compressed_size (codec, instat.st_size);
902  } else if (codec->funcs.get_uncompressed_size != NULL) {
903  outsize_max = squash_codec_get_uncompressed_size (codec, inbuf, instat.st_size);
904  } else {
905  outsize_max = instat.st_size - 1;
906  outsize_max |= outsize_max >> 1;
907  outsize_max |= outsize_max >> 2;
908  outsize_max |= outsize_max >> 4;
909  outsize_max |= outsize_max >> 8;
910  outsize_max |= outsize_max >> 16;
911  outsize_max++;
912 
913  outsize_max <<= 2;
914  }
915 
916  if (ftruncate(outfd, (off_t) outsize_max) != -1) {
917  if ((outbuf = mmap (NULL, outsize_max, PROT_WRITE | PROT_READ, MAP_SHARED, outfd, 0)) != MAP_FAILED) {
918  size_t outsize = outsize_max;
919 
920  if (stream_type == SQUASH_STREAM_COMPRESS) {
921  res = squash_codec_compress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
922 
923  if (res == SQUASH_OK) {
924  if (ftruncate (outfd, (off_t) outsize) == -1) {
925  res = SQUASH_FAILED;
926  }
927  if (fseek (output, outsize, SEEK_SET) == -1) {
928  res = SQUASH_FAILED;
929  }
930  }
931  } else {
932  res = squash_codec_decompress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
933  while (res == SQUASH_BUFFER_FULL) {
934  munmap (outbuf, outsize_max);
935  outsize_max <<= 1;
936  if ((ftruncate (outfd, (off_t) outsize_max) == -1) ||
937  (outbuf = mmap (NULL, outsize_max, PROT_WRITE | PROT_READ, MAP_SHARED, outfd, 0)) == MAP_FAILED) {
938  break;
939  } else {
940  outsize = outsize_max;
941  res = squash_codec_decompress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
942  }
943  }
944 
945  if (res == SQUASH_OK) {
946  if (ftruncate (outfd, (off_t) outsize) == -1) {
947  res = SQUASH_FAILED;
948  }
949  if (fseek (output, outsize, SEEK_SET) == -1) {
950  res = SQUASH_FAILED;
951  }
952  }
953  }
954 
955  if (outbuf != MAP_FAILED) {
956  munmap (outbuf, outsize_max);
957  }
958  }
959  }
960 
961  munmap (inbuf, instat.st_size);
962  }
963  }
964  }
965  }
966 
967  if (res == SQUASH_OK) {
968  return res;
969  }
970 #endif /* !defined(_WIN32) */
971 
972  stream = squash_codec_create_stream_with_options (codec, stream_type, options);
973  if ( stream == NULL ) {
974  return SQUASH_FAILED;
975  }
976 
977  inbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
978  outbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
979 
980  if (inbuf == NULL || outbuf == NULL) {
981  squash_object_unref (stream);
982  free (inbuf);
983  free (outbuf);
984  return SQUASH_MEMORY;
985  }
986 
987  while ( !feof (input) ) {
988  stream->next_in = inbuf;
989  stream->avail_in = fread (inbuf, 1, SQUASH_CODEC_FILE_BUF_SIZE, input);
990 
991  do {
992  stream->next_out = outbuf;
993  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
994 
995  res = squash_stream_process (stream);
996 
997  fwrite (outbuf, 1, stream->next_out - outbuf, output);
998  } while ( res == SQUASH_PROCESSING );
999 
1000  if ( res != SQUASH_OK ) {
1001  if ( res == SQUASH_END_OF_STREAM && stream_type == SQUASH_STREAM_DECOMPRESS ) {
1002  res = SQUASH_OK;
1003  }
1004  squash_object_unref (stream);
1005  free (inbuf);
1006  free (outbuf);
1007  return res;
1008  }
1009  }
1010 
1011  do {
1012  stream->next_out = outbuf;
1013  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
1014 
1015  res = squash_stream_finish (stream);
1016 
1017  fwrite (outbuf, 1, stream->next_out - outbuf, output);
1018  } while ( res == SQUASH_PROCESSING );
1019 
1020  squash_object_unref (stream);
1021  free (inbuf);
1022  free (outbuf);
1023 
1024  return res;
1025 }
