531 SquashStatus
532 squash_codec_compress_with_options (SquashCodec* codec,
533  size_t* compressed_length,
534  uint8_t compressed[SQUASH_ARRAY_PARAM(*compressed_length)],
535  size_t uncompressed_length,
536  const uint8_t uncompressed[SQUASH_ARRAY_PARAM(uncompressed_length)],
537  SquashOptions* options) {
538  SquashCodecImpl* impl = NULL;
539 
540  assert (codec != NULL);
541 
542  assert (compressed != NULL);
543  assert (uncompressed != NULL);
544 
545  impl = squash_codec_get_impl (codec);
546  if (impl == NULL)
547  return squash_error (SQUASH_UNABLE_TO_LOAD);
548 
549  if (compressed == uncompressed)
550  return squash_error (SQUASH_INVALID_BUFFER);
551 
552  if (impl->compress_buffer ||
553  impl->compress_buffer_unsafe) {
554  size_t max_compressed_length = squash_codec_get_max_compressed_size (codec, uncompressed_length);
555 
556  if (*compressed_length >= max_compressed_length) {
557  if (impl->compress_buffer_unsafe != NULL) {
558  return impl->compress_buffer_unsafe (codec,
559  compressed_length, compressed,
560  uncompressed_length, uncompressed,
561  options);
562  } else {
563  return impl->compress_buffer (codec,
564  compressed_length, compressed,
565  uncompressed_length, uncompressed,
566  options);
567  }
568  } else if (impl->compress_buffer != NULL) {
569  return impl->compress_buffer (codec,
570  compressed_length, compressed,
571  uncompressed_length, uncompressed,
572  options);
573  } else {
574  SquashStatus status;
575  uint8_t* tmp_buf = malloc (max_compressed_length);
576  if (tmp_buf == NULL)
577  return squash_error (SQUASH_MEMORY);
578 
579  status = impl->compress_buffer_unsafe (codec,
580  &max_compressed_length, tmp_buf,
581  uncompressed_length, uncompressed,
582  options);
583  if (status == SQUASH_OK) {
584  if (*compressed_length < max_compressed_length) {
585  *compressed_length = max_compressed_length;
586  free (tmp_buf);
587  return squash_error (SQUASH_BUFFER_FULL);
588  } else {
589  *compressed_length = max_compressed_length;
590  memcpy (compressed, tmp_buf, max_compressed_length);
591  free (tmp_buf);
592  return SQUASH_OK;
593  }
594  } else {
595  free (tmp_buf);
596  return status;
597  }
598  }
599  } else {
600  SquashStatus status;
601  SquashStream* stream;
602 
603  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_COMPRESS, options);
604  if (stream == NULL)
605  return squash_error (SQUASH_FAILED);
606 
607  stream->next_in = uncompressed;
608  stream->avail_in = uncompressed_length;
609  stream->next_out = compressed;
610  stream->avail_out = *compressed_length;
611 
612  do {
613  status = squash_stream_process (stream);
614  } while (status == SQUASH_PROCESSING);
615 
616  if (status != SQUASH_OK) {
617  squash_object_unref (stream);
618  return status;
619  }
620 
621  do {
622  status = squash_stream_finish (stream);
623  } while (status == SQUASH_PROCESSING);
624 
625  if (status != SQUASH_OK) {
626  squash_object_unref (stream);
627  return status;
628  }
629 
630  *compressed_length = stream->total_out;
631  squash_object_unref (stream);
632  return status;
633  }
634 }

684 SquashStatus
685 squash_codec_decompress_with_options (SquashCodec* codec,
686  size_t* decompressed_length,
687  uint8_t decompressed[SQUASH_ARRAY_PARAM(*decompressed_length)],
688  size_t compressed_length,
689  const uint8_t compressed[SQUASH_ARRAY_PARAM(compressed_length)],
690  SquashOptions* options) {
691  SquashCodecImpl* impl = NULL;
692 
693  assert (codec != NULL);
694 
695  impl = squash_codec_get_impl (codec);
696  if (impl == NULL)
697  return squash_error (SQUASH_UNABLE_TO_LOAD);
698 
699  if (decompressed == compressed)
700  return squash_error (SQUASH_INVALID_BUFFER);
701 
702  if (impl->decompress_buffer != NULL) {
703  return impl->decompress_buffer (codec,
704  decompressed_length, decompressed,
705  compressed_length, compressed,
706  options);
707  } else {
708  SquashStatus status;
709  SquashStream* stream;
710 
711  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_DECOMPRESS, options);
712  stream->next_in = compressed;
713  stream->avail_in = compressed_length;
714  stream->next_out = decompressed;
715  stream->avail_out = *decompressed_length;
716 
717  do {
718  status = squash_stream_process (stream);
719  } while (status == SQUASH_PROCESSING);
720 
721  if (status == SQUASH_END_OF_STREAM) {
722  status = SQUASH_OK;
723  *decompressed_length = stream->total_out;
724  } else if (status == SQUASH_OK) {
725  do {
726  status = squash_stream_finish (stream);
727  } while (status == SQUASH_PROCESSING);
728 
729  if (status == SQUASH_OK) {
730  *decompressed_length = stream->total_out;
731  }
732  }
733 
734  assert (stream->stream_type == SQUASH_STREAM_DECOMPRESS);
735  squash_object_unref (stream);
736 
737  return status;
738  }
739 }

979 static SquashStatus
980 squash_codec_process_file_with_options (SquashCodec* codec,
981  SquashStreamType stream_type,
982  FILE* output,
983  size_t output_length,
984  FILE* input,
985  SquashOptions* options) {
986  SquashStatus res = SQUASH_FAILED;
987  SquashStream* stream;
988 
989  if (codec->impl.process_stream == NULL) {
990  /* Attempt to mmap the input and output. This short circuits the
991  whole SquashBufferStream hack when possible, which can be a
992  huge performance boost (50-100% for several codecs). */
993  SquashMappedFile* in_map;
994 
995  in_map = squash_mapped_file_new (input, false);
996  if (in_map != NULL) {
997  SquashMappedFile* out_map;
998  size_t max_output_size;
999 
1000  if (stream_type == SQUASH_STREAM_COMPRESS) {
1001  max_output_size = squash_codec_get_max_compressed_size (codec, in_map->data_length);
1002  } else if (codec->impl.get_uncompressed_size != NULL) {
1003  max_output_size = squash_codec_get_uncompressed_size (codec, in_map->data_length, in_map->data);
1004  } else if (output_length != 0) {
1005  max_output_size = output_length;
1006  } else {
1007  max_output_size = in_map->data_length - 1;
1008  max_output_size |= max_output_size >> 1;
1009  max_output_size |= max_output_size >> 2;
1010  max_output_size |= max_output_size >> 4;
1011  max_output_size |= max_output_size >> 8;
1012  max_output_size |= max_output_size >> 16;
1013  max_output_size++;
1014 
1015  max_output_size <<= 3;
1016  }
1017 
1018  out_map = squash_mapped_file_new_full(output, false, 0, max_output_size);
1019  if (out_map != NULL) {
1020  size_t output_size = out_map->data_length;
1021 
1022  if (stream_type == SQUASH_STREAM_COMPRESS) {
1023  output_size = out_map->data_length;
1024  res = squash_codec_compress_with_options (codec, &output_size, out_map->data, in_map->data_length, in_map->data, options);
1025  } else {
1026  do {
1027  output_size = max_output_size;
1028  res = squash_codec_decompress_with_options (codec, &output_size, out_map->data, in_map->data_length, in_map->data, options);
1029  max_output_size <<= 1;
1030  } while (SQUASH_UNLIKELY(res == SQUASH_BUFFER_FULL) && output_length == 0 && codec->impl.get_uncompressed_size == NULL);
1031  }
1032 
1033  squash_mapped_file_free (out_map);
1034  out_map = NULL;
1035 
1036  /* We know that the mmap was successful, so not checking
1037  fileno() should be safe. */
1038  if (ftruncate (fileno (output), (off_t) output_size) == -1)
1039  res = SQUASH_FAILED;
1040  if (fseek (output, output_size, SEEK_SET) == -1)
1041  res = SQUASH_FAILED;
1042  }
1043 
1044  if (res != SQUASH_OK) {
1045  size_t ipos = in_map->data_offset;
1046  squash_mapped_file_free (in_map);
1047  fseek (input, ipos, SEEK_SET);
1048  } else {
1049  squash_mapped_file_free (in_map);
1050  }
1051  in_map = NULL;
1052  }
1053 
1054  if (res == SQUASH_OK)
1055  return res;
1056  }
1057 
1058  stream = squash_codec_create_stream_with_options (codec, stream_type, options);
1059  if (stream == NULL) {
1060  return squash_error (SQUASH_FAILED);
1061  }
1062 
1063  uint8_t* inbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
1064  uint8_t* outbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
1065 
1066  if (inbuf == NULL || outbuf == NULL) {
1067  squash_object_unref (stream);
1068  free (inbuf);
1069  free (outbuf);
1070  return squash_error (SQUASH_MEMORY);
1071  }
1072 
1073  while ( !feof (input) ) {
1074  stream->next_in = inbuf;
1075  stream->avail_in = fread (inbuf, 1, SQUASH_CODEC_FILE_BUF_SIZE, input);
1076 
1077  do {
1078  stream->next_out = outbuf;
1079  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
1080 
1081  res = squash_stream_process (stream);
1082 
1083  fwrite (outbuf, 1, stream->next_out - outbuf, output);
1084  } while ( res == SQUASH_PROCESSING );
1085 
1086  if ( res != SQUASH_OK ) {
1087  if ( res == SQUASH_END_OF_STREAM && stream_type == SQUASH_STREAM_DECOMPRESS ) {
1088  res = SQUASH_OK;
1089  }
1090  squash_object_unref (stream);
1091  free (inbuf);
1092  free (outbuf);
1093  return res;
1094  }
1095  }
1096 
1097  do {
1098  stream->next_out = outbuf;
1099  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
1100 
1101  res = squash_stream_finish (stream);
1102 
1103  fwrite (outbuf, 1, stream->next_out - outbuf, output);
1104  } while ( res == SQUASH_PROCESSING );
1105 
1106  squash_object_unref (stream);
1107  free (inbuf);
1108  free (outbuf);
1109 
1110  return res;
1111 }
