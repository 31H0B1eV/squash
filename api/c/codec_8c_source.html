481 SquashStatus
482 squash_codec_compress_with_options (SquashCodec* codec,
483  uint8_t* compressed, size_t* compressed_length,
484  const uint8_t* uncompressed, size_t uncompressed_length,
485  SquashOptions* options) {
486  SquashPlugin* plugin = NULL;
487  SquashCodecFuncs* funcs = NULL;
488 
489  assert (codec != NULL);
490 
491  plugin = codec->plugin;
492  assert (plugin != NULL);
493 
494  funcs = squash_codec_get_funcs (codec);
495  if (funcs == NULL) {
496  return SQUASH_UNABLE_TO_LOAD;
497  } else if (funcs->compress_buffer != NULL) {
498  return funcs->compress_buffer (codec,
499  compressed, compressed_length,
500  uncompressed, uncompressed_length,
501  options);
502  } else {
503  SquashStatus status;
504  SquashStream* stream;
505 
506  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_COMPRESS, options);
507  stream->next_in = uncompressed;
508  stream->avail_in = uncompressed_length;
509  stream->next_out = compressed;
510  stream->avail_out = *compressed_length;
511 
512  do {
513  status = squash_stream_process (stream);
514  } while (status == SQUASH_PROCESSING);
515 
516  if (status != SQUASH_OK) {
517  squash_object_unref (stream);
518  return status;
519  }
520 
521  do {
522  status = squash_stream_finish (stream);
523  } while (status == SQUASH_PROCESSING);
524 
525  if (status != SQUASH_OK) {
526  squash_object_unref (stream);
527  return status;
528  }
529 
530  *compressed_length = stream->total_out;
531  squash_object_unref (stream);
532  return status;
533  }
534 }

854 static SquashStatus
855 squash_codec_process_file_with_options (SquashCodec* codec,
856  SquashStreamType stream_type,
857  FILE* output, FILE* input,
858  SquashOptions* options) {
859  SquashStatus res = SQUASH_FAILED;
860  SquashStream* stream;
861  uint8_t* inbuf = NULL;
862  uint8_t* outbuf = NULL;
863 
864  if (codec->funcs.create_stream == NULL) {
865  /* Attempt to mmap the input and output. This short circuits the
866  whole SquashBufferStream hack when possible, which can be a
867  huge performance boost (50-100% for several codecs). */
868  struct stat instat = { 0, };
869  size_t outsize_max = 0;
870  int infd = fileno (input);
871  int outfd = fileno (output);
872 
873  if (infd > 0) {
874  if (fstat (infd, &instat) != -1) {
875  if ((inbuf = mmap (NULL, instat.st_size, PROT_READ, MAP_SHARED, infd, 0)) != MAP_FAILED) {
876  if (stream_type == SQUASH_STREAM_COMPRESS) {
877  outsize_max = squash_codec_get_max_compressed_size (codec, instat.st_size);
878  } else if (codec->funcs.get_uncompressed_size != NULL) {
879  outsize_max = squash_codec_get_uncompressed_size (codec, inbuf, instat.st_size);
880  } else {
881  outsize_max = instat.st_size - 1;
882  outsize_max |= outsize_max >> 1;
883  outsize_max |= outsize_max >> 2;
884  outsize_max |= outsize_max >> 4;
885  outsize_max |= outsize_max >> 8;
886  outsize_max |= outsize_max >> 16;
887  outsize_max++;
888 
889  outsize_max <<= 1;
890  }
891 
892  if (ftruncate(outfd, (off_t) outsize_max) != -1) {
893  if ((outbuf = mmap (NULL, outsize_max, PROT_WRITE | PROT_READ, MAP_SHARED, outfd, 0)) != MAP_FAILED) {
894  size_t outsize = outsize_max;
895 
896  if (stream_type == SQUASH_STREAM_COMPRESS) {
897  res = squash_codec_compress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
898 
899  if (res == SQUASH_OK) {
900  ftruncate (outfd, (off_t) outsize);
901  fseek (output, outsize, SEEK_SET);
902  }
903  } else {
904  res = squash_codec_decompress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
905  while (res == SQUASH_BUFFER_FULL) {
906  munmap (outbuf, outsize_max);
907  outsize_max <<= 1;
908  if ((ftruncate (outfd, (off_t) outsize_max) == -1) ||
909  (outbuf = mmap (NULL, outsize_max, PROT_WRITE | PROT_READ, MAP_SHARED, outfd, 0)) == MAP_FAILED) {
910  break;
911  } else {
912  outsize = outsize_max;
913  res = squash_codec_decompress_with_options (codec, outbuf, &outsize, inbuf, instat.st_size, options);
914  }
915  }
916 
917  if (res == SQUASH_OK) {
918  ftruncate (outfd, (off_t) outsize);
919  }
920  }
921 
922  if (outbuf != MAP_FAILED) {
923  munmap (outbuf, outsize_max);
924  }
925  }
926  }
927 
928  munmap (inbuf, instat.st_size);
929  }
930  }
931  }
932  }
933 
934  if (res == SQUASH_OK) {
935  return res;
936  }
937 
938  stream = squash_codec_create_stream_with_options (codec, stream_type, options);
939  if ( stream == NULL ) {
940  return SQUASH_FAILED;
941  }
942 
943  inbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
944  outbuf = malloc (SQUASH_CODEC_FILE_BUF_SIZE);
945 
946  if (inbuf == NULL || outbuf == NULL) {
947  squash_object_unref (stream);
948  free (inbuf);
949  free (outbuf);
950  return SQUASH_MEMORY;
951  }
952 
953  while ( !feof (input) ) {
954  stream->next_in = inbuf;
955  stream->avail_in = fread (inbuf, 1, SQUASH_CODEC_FILE_BUF_SIZE, input);
956 
957  do {
958  stream->next_out = outbuf;
959  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
960 
961  res = squash_stream_process (stream);
962 
963  fwrite (outbuf, 1, stream->next_out - outbuf, output);
964  } while ( res == SQUASH_PROCESSING );
965 
966  if ( res != SQUASH_OK ) {
967  if ( res == SQUASH_END_OF_STREAM && stream_type == SQUASH_STREAM_DECOMPRESS ) {
968  res = SQUASH_OK;
969  }
970  squash_object_unref (stream);
971  free (inbuf);
972  free (outbuf);
973  return res;
974  }
975  }
976 
977  do {
978  stream->next_out = outbuf;
979  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
980 
981  res = squash_stream_finish (stream);
982 
983  fwrite (outbuf, 1, stream->next_out - outbuf, output);
984  } while ( res == SQUASH_PROCESSING );
985 
986  squash_object_unref (stream);
987  free (inbuf);
988  free (outbuf);
989 
990  return res;
991 }
