98 squash_plugin_init (SquashPlugin* plugin) {
99  if (plugin->plugin == NULL) {
100  lt_dlhandle handle;
101  char* plugin_file_name;
102  size_t plugin_dir_length;
103  size_t plugin_name_length;
104  size_t plugin_file_name_max_length;
105  size_t squash_version_api_length = strlen (SQUASH_VERSION_API);
106 
107  plugin_dir_length = strlen (plugin->directory);
108  plugin_name_length = strlen (plugin->name);
109  plugin_file_name_max_length = plugin_dir_length + squash_version_api_length + plugin_name_length + 19;
110  plugin_file_name = (char*) malloc (plugin_file_name_max_length + 1);
111 
112  snprintf (plugin_file_name, plugin_file_name_max_length + 1,
113  "%s/libsquash%s-plugin-%s", plugin->directory, SQUASH_VERSION_API, plugin->name);
114 
115  lt_dlinit ();
116  handle = lt_dlopenext (plugin_file_name);
117  free (plugin_file_name);
118 
119  if (handle != NULL) {
120  static pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
121  pthread_mutex_lock (&mutex);
122  if (plugin->plugin == NULL) {
123  plugin->plugin = handle;
124  handle = NULL;
125  }
126  pthread_mutex_unlock (&mutex);
127  }
128 
129  if (handle != NULL) {
130  lt_dlclose (handle);
131  } else {
132  SquashStatus (*init_func) (SquashPlugin*);
133  *(void **) (&init_func) = lt_dlsym (plugin->plugin, "squash_plugin_init");
134 
135  if (init_func != NULL) {
136  init_func (plugin);
137  }
138  }
139  }
140 
141  return (plugin->plugin != NULL) ? SQUASH_OK : SQUASH_UNABLE_TO_LOAD;
142 }

206 squash_plugin_init_codec (SquashPlugin* plugin, SquashCodec* codec, SquashCodecFuncs* funcs) {
207  SquashStatus res = SQUASH_OK;
208 
209  assert (plugin != NULL);
210 
211  if (plugin->plugin == NULL) {
212  res = squash_plugin_init (plugin);
213  if (res != SQUASH_OK) {
214  return res;
215  }
216  }
217 
218  if (codec->initialized == 0) {
219  SquashStatus (*init_codec_func) (SquashCodec*, SquashCodecFuncs*);
220  static pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
221 
222  *(void **) (&init_codec_func) = lt_dlsym (plugin->plugin, "squash_plugin_init_codec");
223 
224  if (init_codec_func == NULL) {
225  return SQUASH_UNABLE_TO_LOAD;
226  }
227 
228  pthread_mutex_lock (&mutex);
229  res = init_codec_func (codec, funcs);
230  codec->initialized = (res == SQUASH_OK);
231  pthread_mutex_unlock (&mutex);
232  }
233 
234  return res;
235 }
