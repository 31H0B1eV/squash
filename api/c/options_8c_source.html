67 squash_options_parse_option (SquashOptions* options, const char* key, const char* value) {
68  SquashCodecFuncs* funcs;
69 
70  assert (options != NULL);
71  assert (options->codec != NULL);
72 
73  if (key == NULL) {
74  return SQUASH_OK;
75  }
76 
77  funcs = squash_codec_get_funcs (options->codec);
78  assert (funcs != NULL);
79 
80  if (funcs->parse_option != NULL) {
81  return funcs->parse_option (options, key, value);
82  } else {
83  return (strcasecmp (key, "level") == 0) ? SQUASH_OK : SQUASH_BAD_PARAM;
84  }
85 }

96 squash_options_parsea (SquashOptions* options, const char* const* keys, const char* const* values) {
97  SquashStatus status = SQUASH_OK;
98  int n;
99 
100  assert (options != NULL);
101 
102  if (keys == NULL || values == NULL)
103  return SQUASH_OK;
104 
105  for ( n = 0 ; keys[n] != NULL ; n++ ) {
106  status = squash_options_parse_option (options, keys[n], values[n]);
107  if (status != SQUASH_OK) {
108  break;
109  }
110  }
111 
112  return status;
113 }
