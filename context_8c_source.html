201 squash_context_add_codec (SquashContext* context, SquashCodec* codec) {
202  SquashCodecRef* codec_ref;
203 
204  assert (context != NULL);
205  assert (codec != NULL);
206 
207  codec_ref = squash_context_get_codec_ref (context, codec->name);
208  if (codec_ref == NULL) {
209  /* Insert a new entry into context->codecs */
210  codec_ref = (SquashCodecRef*) malloc (sizeof (SquashCodecRef));
211  codec_ref->codec = codec;
212  SQUASH_TREE_ENTRY_INIT(codec_ref->tree);
213  SQUASH_TREE_INSERT (&(context->codecs), SquashCodecRef_s, tree, codec_ref);
214  } else if (codec->priority > codec_ref->codec->priority) {
215  /* Switch the existing context codec's details to this one */
216  codec_ref->codec = codec;
217  }

220 static void
221 squash_context_find_plugins_in_directory (SquashContext* context, const char* directory_name) {
222  DIR* directory = opendir (directory_name);
223  size_t directory_name_length = strlen (directory_name);
224  struct dirent* result = NULL;
225  struct dirent* entry = NULL;
226 
227  if (directory == NULL) {
228  return;
229  } else {
230  long name_max = pathconf (directory_name, _PC_NAME_MAX);
231  if (name_max == -1)
232  name_max = 255;
233  entry = (struct dirent*) malloc (offsetof (struct dirent, d_name) + name_max + 1);
234  }
235 
236  while ( readdir_r (directory, entry, &result) == 0 && result != NULL ) {
237 #ifdef _DIRENT_HAVE_D_TYPE
238  if ( entry->d_type != DT_DIR &&
239  entry->d_type != DT_UNKNOWN &&
240  entry->d_type != DT_LNK )
241  continue;
242 #endif
243 
244  if (strcmp (entry->d_name, "..") == 0 ||
245  strcmp (entry->d_name, ".") == 0)
246  continue;
247 
248  size_t plugin_name_length = strlen (entry->d_name);
249  char* plugin_name = entry->d_name;
250 
251  size_t codecs_file_name_length = directory_name_length + (plugin_name_length * 2) + 10;
252  char* codecs_file_name = (char*) malloc (codecs_file_name_length + 1);
253  snprintf (codecs_file_name, codecs_file_name_length, "%s/%s/%s.codecs",
254  directory_name, plugin_name, plugin_name);
255 
256  FILE* codecs_file = fopen (codecs_file_name, "r");
257 
258  if (codecs_file != NULL) {
259  size_t plugin_directory_name_length = directory_name_length + plugin_name_length + 1;
260  char* plugin_directory_name = (char*) malloc (plugin_directory_name_length + 1);
261  snprintf (plugin_directory_name, plugin_directory_name_length + 1, "%s/%s",
262  directory_name, plugin_name);
263 
264  SquashPlugin* plugin = squash_context_add_plugin (context, strdup (plugin_name), plugin_directory_name);
265  if (plugin != NULL) {
266  char line[256] = { 0, };
267 
268  while (fgets (line, 255, codecs_file) != NULL) {
269  int line_len;
270  for ( line_len = (int) strlen (line) - 1 ; line_len >= 0 ; line_len-- ) {
271  if (isspace (line[line_len]))
272  line[line_len] = 0;
273  else
274  break;
275  }
276 
277  char* priority = strchr (line, ' ');
278  if (priority != NULL) {
279  *priority = 0;
280  priority++;
281  }
282 
283  if (*line != 0) {
284  unsigned int priority_value = 0;
285 
286  if (priority != NULL) {
287  char* priority_end = NULL;
288  priority_value = strtol (priority, &priority_end, 0);
289  if (*priority_end != 0)
290  priority_value = 0;
291  } else {
292  priority_value = 50;
293  }
294 
295  squash_codec_new (strdup (line), priority_value, plugin);
296  }
297  }
298  }
299 
300  fclose (codecs_file);
301  }
302 
303  free (codecs_file_name);
304  }
305 
306  free (entry);
307  closedir (directory);
308 }

310 static void
311 squash_context_find_plugins (SquashContext* context) {
312  char* directories;
313 
314  assert (context != NULL);
315 
316  directories = getenv ("SQUASH_PLUGINS");
317 
318  if (directories != NULL) {
319  char* saveptr = NULL;
320  char* directory_name = NULL;
321 
322  directories = strdup (directories);
323 
324  for ( directory_name = strtok_r (directories, ":", &saveptr) ;
325  directory_name != NULL ;
326  directory_name = strtok_r (NULL, ":", &saveptr) ) {
327  squash_context_find_plugins_in_directory (context, directory_name);
328  }
329 
330  free (directories);
331  } else {
332  squash_context_find_plugins_in_directory (context, SQUASH_PLUGIN_DIR);
333  }
334 }

464 SquashContext*
465 squash_context_get_default (void) {
466  static volatile SquashContext* squash_context_default;
467  static volatile unsigned int status = 0;
468 
469  if (status == 0) {
470  if (__sync_bool_compare_and_swap (&status, 0, 1)) {
471  squash_context_default = (SquashContext*) squash_context_new ();
472  status = 2;
473  __sync_synchronize ();
474  }
475  }
476 
477  while (status != 2) { /* Spin */ }
478 
479  return squash_context_default;
480 }
