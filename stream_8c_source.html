325 SquashStatus
326 squash_stream_process (SquashStream* stream) {
327  SquashCodec* codec;
328  SquashCodecFuncs* funcs;
329  SquashStatus res;
330 
331  assert (stream != NULL);
332 
333  switch (stream->state) {
334  case SQUASH_STREAM_STATE_IDLE:
335  case SQUASH_STREAM_STATE_RUNNING:
336  break;
337  default:
338  return SQUASH_INVALID_OPERATION;
339  }
340 
341  codec = stream->codec;
342  assert (codec != NULL);
343 
344  if (stream->avail_out == 0) {
345  return SQUASH_BUFFER_FULL;
346  }
347 
348  funcs = squash_codec_get_funcs (codec);
349 
350  const size_t avail_in = stream->avail_in;
351  const size_t avail_out = stream->avail_out;
352 
353  if (funcs->process_stream != NULL) {
354  res = funcs->process_stream (stream);
355  } else if (funcs->create_stream == NULL && funcs->flush_stream == NULL && funcs->finish_stream == NULL) {
356  res = squash_buffer_stream_process ((SquashBufferStream*) stream);
357  } else {
358  res = SQUASH_INVALID_OPERATION;
359  }
360 
361  stream->total_in += (avail_in - stream->avail_in);
362  stream->total_out += (avail_out - stream->avail_out);
363 
364  switch (res) {
365  case SQUASH_OK:
366  case SQUASH_PROCESSING:
367  stream->state = SQUASH_STREAM_STATE_RUNNING;
368  break;
369  case SQUASH_END_OF_STREAM:
370  assert (stream->avail_in == 0);
371  stream->state = SQUASH_STREAM_STATE_FINISHED;
372  break;
373  }
374 
375  return res;
376 }

390  SquashCodec* codec;

392  SquashStatus res;
393 
394  assert (stream != NULL);

401  default:

403  }

405  if (stream->stream_type != SQUASH_STREAM_COMPRESS) {
406  return SQUASH_INVALID_OPERATION;

408

410  assert (codec != NULL);
411 
412  funcs = squash_codec_get_funcs (codec);
413 
414  if (funcs != NULL && funcs->flush_stream != NULL) {
415  const size_t avail_in = stream->avail_in;
416  const size_t avail_out = stream->avail_out;
417 
418  res = funcs->flush_stream (stream);
419 
420  stream->total_in += (avail_in - stream->avail_in);
421  stream->total_out += (avail_out - stream->avail_out);
422  } else {
423  res = SQUASH_INVALID_OPERATION;
424  }

448  SquashStatus res;

461  }
462 
463  codec = stream->codec;
464  assert (codec != NULL);
465 
466  funcs = squash_codec_get_funcs (codec);
467 
468  const size_t avail_in = stream->avail_in;
469  const size_t avail_out = stream->avail_out;
470 
471  if (funcs->finish_stream != NULL) {
472  res = funcs->finish_stream (stream);
473  } else if (funcs->create_stream == NULL && funcs->process_stream == NULL && funcs->flush_stream == NULL) {
474  res = squash_buffer_stream_finish ((SquashBufferStream*) stream);
475  } else {
476  return SQUASH_OK;
477  }
478 
479  stream->total_in += (avail_in - stream->avail_in);
480  stream->total_out += (avail_out - stream->avail_out);
481 
482  switch (res) {
483  case SQUASH_OK:
484  case SQUASH_END_OF_STREAM:
485  stream->state = SQUASH_STREAM_STATE_FINISHED;
486  break;
487  case SQUASH_PROCESSING:
488  stream->state = SQUASH_STREAM_STATE_FINISHING;
489  break;
490  }
491 
492  return res;
493 }
