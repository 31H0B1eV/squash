111 squash_status_to_string (SquashStatus status) {
112  switch (status) {
113  case SQUASH_OK:
114  return "Operation completed successfully";
115  case SQUASH_PROCESSING:
116  return "Operation partially completed";
117  case SQUASH_END_OF_STREAM:
118  return "End of stream reached";
119  case SQUASH_FAILED:
120  return "Operation failed";
121  case SQUASH_UNABLE_TO_LOAD:
122  return "Unable to load the requested resource";
123  case SQUASH_BAD_PARAM:
124  return "One or more of the parameters were not valid";
125  case SQUASH_BAD_VALUE:
126  return "One or more parameter values was not valid";
127  case SQUASH_MEMORY:
128  return "Not enough memory is available";
129  case SQUASH_BUFFER_FULL:
130  return "Insufficient space in buffer";
131  case SQUASH_BUFFER_EMPTY:
132  return "Unable to read from buffer";
133  case SQUASH_STATE:
134  return "Performing the requested operation from the current state is not supported";
135  case SQUASH_INVALID_OPERATION:
136  return "The requested operation is not available";
137  case SQUASH_NOT_FOUND:
138  return "The requested codec could not be found";
139  default:
140  return "Unknown.";
141  }
142 }
