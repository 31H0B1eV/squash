462 SquashStatus
463 squash_codec_compress_with_options (SquashCodec* codec,
464  uint8_t* compressed, size_t* compressed_length,
465  const uint8_t* uncompressed, size_t uncompressed_length,
466  SquashOptions* options) {
467  SquashPlugin* plugin = NULL;
468  SquashCodecFuncs* funcs = NULL;
469 
470  assert (codec != NULL);
471 
472  plugin = codec->plugin;
473  assert (plugin != NULL);
474 
475  funcs = squash_codec_get_funcs (codec);
476  if (funcs == NULL) {
477  return SQUASH_UNABLE_TO_LOAD;
478  } else if (funcs->compress_buffer != NULL) {
479  return funcs->compress_buffer (codec,
480  compressed, compressed_length,
481  uncompressed, uncompressed_length,
482  options);
483  } else {
484  SquashStatus status;
485  SquashStream* stream;
486 
487  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_COMPRESS, options);
488  stream->next_in = uncompressed;
489  stream->avail_in = uncompressed_length;
490  stream->next_out = compressed;
491  stream->avail_out = *compressed_length;
492 
493  do {
494  status = squash_stream_process (stream);
495  } while (status == SQUASH_PROCESSING);
496 
497  if (status != SQUASH_OK) {
498  squash_object_unref (stream);
499  return status;
500  }
501 
502  do {
503  status = squash_stream_finish (stream);
504  } while (status == SQUASH_PROCESSING);
505 
506  if (status != SQUASH_OK) {
507  squash_object_unref (stream);
508  return status;
509  }
510 
511  *compressed_length = stream->total_out;
512  squash_object_unref (stream);
513  return status;
514  }
515 }

564  uint8_t* decompressed, size_t* decompressed_length,

582  options);
583  } else {
584  SquashStatus status;
585  SquashStream* stream;
586 
587  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_DECOMPRESS, options);
588  stream->next_in = compressed;
589  stream->avail_in = compressed_length;
590  stream->next_out = decompressed;
591  stream->avail_out = *decompressed_length;
592 
593  do {
594  status = squash_stream_process (stream);
595  } while (status == SQUASH_PROCESSING);
596 
597  switch (status) {
598  case SQUASH_END_OF_STREAM:
599  status = SQUASH_OK;
600  case SQUASH_OK:
601  *decompressed_length = stream->total_out;
602  default:
603  squash_object_unref (stream);
604  }
605 
606  return status;
607  }
