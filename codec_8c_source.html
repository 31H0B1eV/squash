470 SquashStatus
471 squash_codec_compress_with_options (SquashCodec* codec,
472  uint8_t* compressed, size_t* compressed_length,
473  const uint8_t* uncompressed, size_t uncompressed_length,
474  SquashOptions* options) {
475  SquashPlugin* plugin = NULL;
476  SquashCodecFuncs* funcs = NULL;
477 
478  assert (codec != NULL);
479 
480  plugin = codec->plugin;
481  assert (plugin != NULL);
482 
483  funcs = squash_codec_get_funcs (codec);
484  if (funcs == NULL) {
485  return SQUASH_UNABLE_TO_LOAD;
486  } else if (funcs->compress_buffer != NULL) {
487  return funcs->compress_buffer (codec,
488  compressed, compressed_length,
489  uncompressed, uncompressed_length,
490  options);
491  } else {
492  SquashStatus status;
493  SquashStream* stream;
494 
495  stream = squash_codec_create_stream_with_options (codec, SQUASH_STREAM_COMPRESS, options);
496  stream->next_in = uncompressed;
497  stream->avail_in = uncompressed_length;
498  stream->next_out = compressed;
499  stream->avail_out = *compressed_length;
500 
501  do {
502  status = squash_stream_process (stream);
503  } while (status == SQUASH_PROCESSING);
504 
505  if (status != SQUASH_OK) {
506  squash_object_unref (stream);
507  return status;
508  }
509 
510  do {
511  status = squash_stream_finish (stream);
512  } while (status == SQUASH_PROCESSING);
513 
514  if (status != SQUASH_OK) {
515  squash_object_unref (stream);
516  return status;
517  }
518 
519  *compressed_length = stream->total_out;
520  squash_object_unref (stream);
521  return status;
522  }
523 }

806 static SquashStatus
807 squash_codec_process_file_with_options (SquashCodec* codec,
808  SquashStreamType stream_type,
809  FILE* output, FILE* input,
810  SquashOptions* options) {
811  SquashStatus res;
812  SquashStream* stream;
813  uint8_t inbuf[SQUASH_CODEC_FILE_BUF_SIZE] = { 0, };
814  uint8_t outbuf[SQUASH_CODEC_FILE_BUF_SIZE] = { 0, };
815 
816  stream = squash_codec_create_stream_with_options (codec, stream_type, options);
817  if ( stream == NULL ) {
818  return SQUASH_FAILED;
819  }
820 
821  while ( !feof (input) ) {
822  stream->next_in = inbuf;
823  stream->avail_in = fread (inbuf, 1, SQUASH_CODEC_FILE_BUF_SIZE, input);
824 
825  do {
826  stream->next_out = outbuf;
827  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
828 
829  res = squash_stream_process (stream);
830 
831  fwrite (outbuf, 1, stream->next_out - outbuf, output);
832  } while ( res == SQUASH_PROCESSING );
833 
834  if ( res != SQUASH_OK ) {
835  if ( res == SQUASH_END_OF_STREAM && stream_type == SQUASH_STREAM_DECOMPRESS ) {
836  res = SQUASH_OK;
837  }
838  squash_object_unref (stream);
839  return res;
840  }
841  }
842 
843  do {
844  stream->next_out = outbuf;
845  stream->avail_out = SQUASH_CODEC_FILE_BUF_SIZE;
846 
847  res = squash_stream_finish (stream);
848 
849  fwrite (outbuf, 1, stream->next_out - outbuf, output);
850  } while ( res == SQUASH_PROCESSING );
851 
852  squash_object_unref (stream);
853 
854  return res;
855 }
